try:
    conn = psycopg2.connect(
        host=os.getenv("DB_HOST"),
        database=os.getenv("DB_NAME"),
        user=os.getenv("DB_USER"),
        password=os.getenv("DB_PASSWORD"),
        port=os.getenv("DB_PORT") or 5432,
    )
    conn.autocommit = False
    logging.info("PostgreSQL connection established.")
except Exception as e:
    logging.exception("Error connecting to the database: %s", e)
    raise

# ---------- Helpers ----------
def fetch_one(query: str, params: tuple = ()):
    with conn.cursor(cursor_factory=psycopg2.extras.RealDictCursor) as cur:
        cur.execute(query, params)
        return cur.fetchone()

def fetch_all(query: str, params: tuple = ()):
    with conn.cursor(cursor_factory=psycopg2.extras.RealDictCursor) as cur:
        cur.execute(query, params)
        return cur.fetchall()

def execute(query: str, params: tuple = ()):
    with conn.cursor() as cur:
        cur.execute(query, params)
        conn.commit()
        return cur.rowcount

def insert_returning(query: str, params: tuple = ()):
    with conn.cursor(cursor_factory=psycopg2.extras.RealDictCursor) as cur:
        cur.execute(query, params)
        try:
            return cur.fetchone()
        except Exception:
            return None
        
VALID_PLANS = {
    "monthly": 1,
    "quarterly": 3,
    "half yearly": 6,
    "yearly": 12
}

@mcp.tool(
    name="check_user_status",
    title="Check User Status by Phone Number",
    description=(
        "Use this tool to check whether a phone number belongs to a valid user. "
        "The tool searches across the following tables: existing_user_details, new_customers, and plan_extension. "
        "It returns a meaningful message describing the user's current status in the system."
    )
)
def check_user_status(contact_number: str):

    if len(contact_number) != 10 or not contact_number.isdigit():
        return {
            "statusCode": 400,
            "user_type": "invalid_input",
            "message": "Invalid contact number format. Please provide a valid 10-digit phone number."
        }

    try:

        # ---- Check existing subscriber ----
        existing_user = fetch_one(
            """
            SELECT *
            FROM existing_user_details
            WHERE contact_number = %s
            """,
            (contact_number,)
        )

        if existing_user:

            # ---- Check renewal request ----
            renewal_request = fetch_one(
                """
                SELECT *
                FROM plan_extension
                WHERE contact_number = %s
                AND status = 'pending'
                ORDER BY id DESC
                LIMIT 1
                """,
                (contact_number,)
            )

            return {
                "statusCode": 200,
                "user_type": "existing_subscriber",
                "user_details": existing_user,
                "renewal_request": renewal_request,
                "message": "Existing subscriber found."
            }

        # ---- Check pending new subscription ----
        new_customer = fetch_one(
            """
            SELECT *
            FROM new_customers
            WHERE contact_number = %s
            """,
            (contact_number,)
        )

        if new_customer:
            return {
                "statusCode": 200,
                "user_type": "new_subscription_request",
                "user_details": new_customer,
                "message": "A new subscription request already exists."
            }

        # ---- Not found anywhere ----
        return {
            "statusCode": 404,
            "user_type": "not_found",
            "message": "Phone number not registered in the system."
        }

    except Exception as e:
        logging.exception("Error checking user status")
        return {
            "statusCode": 500,
            "error": str(e)
        }

@mcp.tool(
    name="note_subscription_request",
    title="Note New Subscription Request",
    description=(
        "Use this tool when a user wants to subscribe to the newspaper. "
        "This tool updates the user's plan and start date in the new_user_details table. "
        "Required inputs: user_name, phone_number, intended_plan, start_date. "
        "The intended_plan MUST be one of the valid plans. "
        "The start_date must be today's date or a future date."
    )
)
def note_subscription_request(
    user_name: str,
    phone_number: str,
    intended_plan: str,
    start_date: str
):
    if not all([user_name, phone_number, intended_plan, start_date]):
        return {
            "statusCode": 400,
            "message": "Missing required fields. Please provide user_name, phone_number, intended_plan, and start_date."
        }

    if len(phone_number) != 10 or not phone_number.isdigit():
        return {
            "statusCode": 400,
            "message": "Invalid phone number format. Please provide a valid 10-digit phone number."
        }

    try:
        plan = intended_plan.strip().lower()

        if plan not in VALID_PLANS:
            return {
                "statusCode": 400,
                "message": f"Invalid plan. Allowed plans are: {list(VALID_PLANS.keys())}"
            }

        try:
            start_dt = datetime.strptime(start_date, "%Y-%m-%d").date()
        except ValueError:
            return {
                "statusCode": 400,
                "message": "Invalid start_date format. Please provide date in YYYY-MM-DD format."
            }

        today = datetime.now(ZoneInfo("Asia/Kolkata")).date()

        if start_dt < today:
            return {
                "statusCode": 400,
                "message": f"Invalid start_date. It must be today's date or a future date. Today's date is {today}."
            }

        existing_user = fetch_one(
            """
            SELECT user_name, phone_number
            FROM new_user_details
            WHERE user_name = %s AND phone_number = %s
            """,
            (user_name, phone_number)
        )

        if not existing_user:
            return {
                "statusCode": 404,
                "message": "User not found in new_user_details."
            }

        query = """
        UPDATE new_user_details
        SET plan = %s,
            start_date = %s
        WHERE user_name = %s
          AND phone_number = %s
        RETURNING user_name, phone_number, plan, start_date;
        """

        result = insert_returning(
            query,
            (plan, start_dt, user_name, phone_number)
        )
        conn.commit()

        return {
            "statusCode": 200,
            "message": "Subscription request updated successfully in new_user_details.",
            "data": result
        }

    except Exception as e:
        conn.rollback()
        logging.exception("Error updating subscription request")
        return {"statusCode": 500, "error": str(e)}

@mcp.tool(
    name="send_renewal_request",
    title="Send Renewal Request",
    description=(
        "Use this tool when an existing subscriber wants to renew or extend their subscription plan. "
        "The tool accepts the subscriber's contact_number and requested plan. "
        "It first verifies the subscriber exists in the existing_user_details table. "
        "If the subscriber exists, their details are automatically fetched and a renewal request is created "
        "in the plan_extension table with status 'pending'."
    )
)
def send_renewal_request(
    contact_number: str,
    req_plan: str
):
    try:
        plan = req_plan.strip().lower()

        if plan not in VALID_PLANS:
            return {
                "statusCode": 400,
                "message": f"Invalid renewal plan. Allowed plans are: {list(VALID_PLANS.keys())}"
            }
        
        if len(contact_number) != 10 or not contact_number.isdigit():
            return {
                "statusCode": 400,
                "message": "Invalid contact number format. Please provide a valid 10-digit phone number."
            }
            
        # Check if user exists 
        user_query = """
        SELECT address
        FROM existing_user_details
        WHERE contact_number = %s;
        """

        user = fetch_one(user_query, (contact_number,))

        if not user:
            return {
                "statusCode": 404,
                "message": "Subscriber not found. Renewal request cannot be created."
            }

        address = user["address"]

        # Calculate extension 
        months_to_add = VALID_PLANS[plan]

        today = datetime.now(ZoneInfo("Asia/Kolkata")).date()
        extended_date = today + relativedelta(months=months_to_add)

        query = """
        INSERT INTO plan_extension
        (contact_number, address, requested_plan, extended_date)
        VALUES (%s, %s, %s, %s)
        RETURNING id, status, extended_date;
        """

        result = insert_returning(
            query,
            (contact_number, address, req_plan, extended_date)
        )
        conn.commit()

        return {
            "statusCode": 200,
            "message": f"Renewal request submitted successfully. New Plan will expire on {extended_date}",
            "renewal_details": result
        }

    except Exception as e:
        conn.rollback()
        logging.exception("Error creating renewal request")
        return {"statusCode": 500, "error": str(e)}

@mcp.tool(
    name="follow_up_user",
    title="Follow Up with Existing User",
    description=(
        "Use this tool to update the follow-up status of an existing user. "
        "This tool updates the user_status field in the new_user_details table. "
        "It requires user_name, phone_number, and a follow-up message from the AI agent."
    )
)
def follow_up_user(
    user_name: str,
    phone_number: str,
    message: str
):
    # Basic validation
    if not all([user_name, phone_number, message]):
        return {
            "statusCode": 400,
            "message": "Missing required fields. Provide user_name, phone_number, and message."
        }

    if len(phone_number) != 10 or not phone_number.isdigit():
        return {
            "statusCode": 400,
            "message": "Invalid phone number format. Must be 10 digits."
        }

    try:
        query = """
        UPDATE new_user_details
        SET user_status = %s
        WHERE user_name = %s
        AND phone_number = %s;
        """

        rows_updated = execute(query, (message, user_name, phone_number))

        if rows_updated == 0:
            return {
                "statusCode": 404,
                "message": "User not found in new_user_details."
            }

        return {
            "statusCode": 200,
            "message": "User follow-up status updated successfully.",
            "updated_rows": rows_updated
        }

    except Exception as e:
        conn.rollback()
        logging.exception("Error updating user follow-up status")
        return {
            "statusCode": 500,
            "error": str(e)
        }
